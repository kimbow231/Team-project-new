    <script>
        // Initialize Lucide icons
        lucide.createIcons();
        
        // --- JavaScript for Page/View Switching ---

        const pages = document.querySelectorAll('.page-content');
        const navItems = document.querySelectorAll('.nav-item');
        const defaultPage = 'login'; // Start on the Login page
        
        // Function to switch the active page
        function switchPage(targetPageId) {
            // 1. Hide all pages
            pages.forEach(page => {
                page.classList.remove('active');
            });

            // 2. Show the target page
            const targetPage = document.querySelector(`[data-page="${targetPageId}"]`);
            if (targetPage) {
                targetPage.classList.add('active');
            }

            // 3. Update navigation bar state
            navItems.forEach(item => {
                const icon = item.querySelector('i');
                const text = item.querySelector('span');
                
                if (item.getAttribute('data-target') === targetPageId) {
                    // Active state styling
                    icon.classList.remove('text-gray-400');
                    icon.classList.add('text-primary-dark');
                    text.classList.remove('text-gray-400');
                    text.classList.add('text-primary-dark', 'font-bold');
                } else {
                    // Inactive state styling
                    icon.classList.add('text-gray-400');
                    icon.classList.remove('text-primary-dark');
                    text.classList.add('text-gray-400');
                    text.classList.remove('text-primary-dark', 'font-bold');
                }
            });
            
            // 4. Hide nav bar on Login/Welcome screen
            const navBar = document.getElementById('nav-bar');
            if (targetPageId === 'login') {
                navBar.classList.add('hidden');
            } else {
                navBar.classList.remove('hidden');
            }
        }
        
        // Add click listeners to navigation items
        navItems.forEach(item => {
            item.addEventListener('click', () => {
                const targetId = item.getAttribute('data-target');
                switchPage(targetId);
            });
        });

        // Initialize the app on the default page
        switchPage(defaultPage);
        
        // If the user navigates past the login screen, set the home button to active visually
        const homeNavButton = document.querySelector('.nav-item[data-target="home"]');
        if (homeNavButton) {
            // This ensures the initial state after 'Continue' button on login is correct
            homeNavButton.click(); 
        }
